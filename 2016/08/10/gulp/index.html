gulp 基于流的自动化构建工具

快速开始

安装和简单配置

  1. 全局安装

    npm i gulp -g

  2. 项目开发依赖

    npm i gulp --save-dev

  3. 创建gulpfile.js文件

    //引入gulp模块
var gulp = require('gulp')
//创建任务
gulp.task('task1',function(){
    //具体任务代码
})

  4. 通过cli来执行任务，不指定任务名默认执行default

    gulp | gulp '任务名'

gulp常用API

gulp.task(name[, deps], fn)

name任务名

deps依赖，格式为数组，依赖会在此任务执行前被执行

fn定义任务所做的一些操作

gulp.src(globs[, options])

要处理文件的入口，可以是单个路径，也可以是数组

gulp.src('./src/index.html')
//多个文件
gulp.src(['index.html','home.html'])

gulp.dest(path[, options])

要生成的文件的路径

gulp.src('./src/less/*.less')
    .pipe(less())
    .pipe(gulp.dest('./dist/css/*.css'))

gulp.watch(glob [, opts], tasks) 或 gulp.watch(glob [, opts, cb])

监听文件变化，执行相应操作

gulp.watch('js/*.js',['task2'])
gulp.watch('js/*.js',function(){})

常用插件

gulp-uglify

js压缩插件

//导入插件
var gulp = require('gulp'),
    uglify= require('gulp-uglify');

//单文件压缩
gulp.task('jsmin', function () {
    gulp.src('src/js/index.js')
        .pipe(uglify())
        .pipe(gulp.dest('dist/js'));
});

//多文件压缩
gulp.task('jsmin', function () {
    gulp.src(['src/js/index.js','src/js/detail.js'])
        .pipe(uglify())
        .pipe(gulp.dest('dist/js'));
});

//排除特定文件
gulp.task('jsmin', function () {
    //压缩src/js目录下的所有js文件
    //除了test1.js和test2.js（**匹配src/js的0个或多个子文件夹）
    gulp.src(['src/js/*.js', '!src/js/**/{test1,test2}.js']) 
        .pipe(uglify())
        .pipe(gulp.dest('dist/js'));
});

//其他参数设置
gulp.task('jsmin', function () {
    gulp.src(['src/js/*.js', '!src/js/**/{test1,test2}.js'])
        .pipe(uglify({
              mangle: {except: ['require' ,'exports' ,'module' ,'$']}//排除混淆关键字
            mangle: true,//类型：Boolean 默认：true 是否修改变量名
            compress: true,//类型：Boolean 默认：true 是否完全压缩
            preserveComments: 'all' //保留所有注释
        }))
        .pipe(gulp.dest('dist/js'));
});

gulp-less gulp-clean-css

less编译和css压缩

//导入插件
var gulp = require('gulp'),
    less = require('gulp-less'),
    cssmin = require('gulp-clean-css')

//less编译和css压缩
gulp.task('testLess', function () {
    //编译src目录下的所有less文件
    //除了reset.less和test.less（**匹配src/less的0个或多个子文件夹）
    //对编译后的css进行压缩
    gulp.src(['src/less/*.less', '!src/less/**/{reset,test}.less']) 
        .pipe(less())
        .pipe(cssmin())
        //.pipe(cssmin({compatibility: 'ie7'}))，兼容IE7
        .pipe(gulp.dest('src/css'));
});

//修改less后自动编译
gulp.task('testWatch', function () {
    gulp.watch('src/**/*.less', ['testLess']); 
  //当所有less文件发生改变时，调用testLess任务
});

//编译异常时会终止watch事件，避免方式如下
var notify = require('gulp-notify'),
    plumber = require('gulp-plumber');

gulp.task('testLess', function () {
    gulp.src('src/less/*.less')
        .pipe(plumber({errorHandler: notify.onError('Error: <%= error.message %>')}))
        .pipe(less())
        .pipe(gulp.dest('src/css'));
});
gulp.task('testWatch', function () {
    gulp.watch('src/**/*.less', ['testLess']);
});

gulp-imagemin

图片压缩插件

var gulp = require('gulp'),
    imagemin = require('gulp-imagemin');
//基本用法和参数设置
gulp.task('testImagemin', function () {
    gulp.src('src/img/*.{png,jpg,gif,ico}')
        .pipe(imagemin({
            optimizationLevel: 5, //类型：Number  默认：3  取值范围：0-7（优化等级）
            progressive: true, //类型：Boolean 默认：false 无损压缩jpg图片
            interlaced: true, //类型：Boolean 默认：false 隔行扫描gif进行渲染
            multipass: true //类型：Boolean 默认：false 多次优化svg直到完全优化
        }))
        .pipe(gulp.dest('dist/img'));
});

//深度压缩
var pngquant = require('imagemin-pngquant');
gulp.task('testImagemin', function () {
    gulp.src('src/img/*.{png,jpg,gif,ico}')
        .pipe(imagemin({
            progressive: true,
            svgoPlugins: [{removeViewBox: false}],//不要移除svg的viewbox属性
            use: [pngquant()] //使用pngquant深度压缩png图片的imagemin插件
        }))
        .pipe(gulp.dest('dist/img'));
});

gulp-concat

js合并插件

var gulp = require('gulp'),
    concat = require('gulp-concat');

gulp.task('testConcat', function () {
    gulp.src('src/js/*.js')
        .pipe(concat('all.js'))//合并后的文件名
        .pipe(gulp.dest('dist/js'));
});

gulp-autoprefixer

css自动处理浏览器前缀的插件，浏览器配置

var gulp = require('gulp'),
    autoprefixer = require('gulp-autoprefixer');

gulp.task('testAutoFx', function () {
    gulp.src('src/css/index.css')
        .pipe(autoprefixer({
            browsers: ['last 2 versions', 'Android >= 4.0'],
            cascade: true, //是否美化属性值 默认：true 像这样：
            //-webkit-transform: rotate(45deg);
            //        transform: rotate(45deg);
            remove:true //是否去掉不必要的前缀 默认：true 
        }))
        .pipe(gulp.dest('dist/css'));
});
